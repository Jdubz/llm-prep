\documentclass[8pt]{article}

% Page geometry — tight margins for maximum content
\usepackage[
  letterpaper,
  margin=0.5in,

  bottom=0.5in
]{geometry}

% Two-column layout
\usepackage{multicol}
\setlength{\columnsep}{0.3in}

\usepackage[T1]{fontenc}

\usepackage{parskip}
