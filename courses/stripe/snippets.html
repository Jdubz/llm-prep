parseCSV
function parseCSV(raw: string) {
  const lines = raw.trim().split("\n");
  const hdr = lines[0].split(",").map(h => h.trim());
  return lines.slice(1).map(l => {
    const v = l.split(",").map(s => s.trim());
    return Object.fromEntries(hdr.map((h, i) => [h, v[i] ?? ""]));
  });
}
parseKV
function parseKV(raw: string): Map<string, string> {
  const m = new Map<string, string>();
  for (const line of raw.trim().split("\n")) {
    const t = line.trim();
    if (!t || t.startsWith("#")) continue;
    const i = t.indexOf("=");
    if (i !== -1) m.set(t.slice(0, i).trim(), t.slice(i + 1).trim());
  }
  return m;
}
groupBy
function groupBy<T>(arr: T[], key: (x: T) => string) {
  const m = new Map<string, T[]>();
  for (const x of arr) {
    const k = key(x);
    if (!m.has(k)) m.set(k, []);
    m.get(k)!.push(x);
  }
  return m;
}
countBy
function countBy<T>(arr: T[], key: (x: T) => string) {
  const m = new Map<string, number>();
  for (const x of arr) {
    const k = key(x);
    m.set(k, (m.get(k) ?? 0) + 1);
  }
  return m;
}
indexBy
function indexBy<T>(arr: T[], key: (x: T) => string) {
  return new Map(arr.map(x => [key(x), x]));
}
dedupBy
function dedupBy<T>(arr: T[], key: (x: T) => string) {
  const seen = new Set<string>();
  return arr.filter(x => { const k = key(x); if (seen.has(k)) return false; seen.add(k); return true; });
}

function sumBy<T>(arr: T[], val: (x: T) => number) {
  return arr.reduce((s, x) => s + val(x), 0);
}

function sortBy<T>(arr: T[], num: (x: T) => number, str: (x: T) => string) {
  return [...arr].sort((a, b) => num(b) - num(a) || str(a).localeCompare(str(b)));
}

balances (from transactions)
function balances(txs: { from: string; to: string; amount: number }[]) {
  const m = new Map<string, number>();
  for (const { from, to, amount } of txs) {
    m.set(from, (m.get(from) ?? 0) - amount);
    m.set(to, (m.get(to) ?? 0) + amount);
  }
  return m;
}
applyTx (with overdraft check)
function applyTx(bals: Map<string, number>, from: string, to: string, amt: number): boolean {
  if (amt <= 0 || from === to) return false;
  const fb = bals.get(from) ?? 0;
  if (fb < amt) return false;
  bals.set(from, fb - amt);
  bals.set(to, (bals.get(to) ?? 0) + amt);
  return true;
}
tieredPrice
function tieredPrice(units: number, tiers: { upTo: number | null; rate: number }[]) {
  let total = 0, prev = 0;
  for (const { upTo, rate } of tiers) {
    const cap = upTo ?? Infinity;
    const qty = Math.min(units, cap) - prev;
    if (qty <= 0) break;
    total += qty * rate;
    prev = cap;
  }
  return total;
}
Rate Limiting
slidingWindowAllow
function slidingWindowAllow(log: Map<string, number[]>, key: string, now: number, max: number, windowMs: number): boolean {
  const ts = log.get(key) ?? [];
  while (ts.length && ts[0] <= now - windowMs) ts.shift();
  if (ts.length >= max) return false;
  ts.push(now);
  log.set(key, ts);
  return true;
}
fixedWindowAllow
function fixedWindowAllow(state: Map<string, { c: number; w: number }>, key: string, now: number, max: number, windowMs: number): boolean {
  const w = Math.floor(now / windowMs);
  const e = state.get(key);
  if (!e || e.w !== w) { state.set(key, { c: 1, w }); return true; }
  if (e.c >= max) return false;
  e.c++;
  return true;
}
String / Parsing
maskPII
function maskPII(s: string) {
  return s
    .replace(/\b\d{4}[\s-]?\d{4}[\s-]?\d{4}[\s-]?\d{4}\b/g, m => "****" + m.slice(-4))
    .replace(/\b[\w.+-]+@[\w-]+\.[\w.-]+\b/g, "[EMAIL]")
    .replace(/\b\d{3}-\d{2}-\d{4}\b/g, "***-**-****");
}
braceExpand
function braceExpand(s: string): string[] {
  const segs: (string | string[])[] = [];
  let i = 0;
  while (i < s.length) {
    if (s[i] === "{") { const j = s.indexOf("}", i); segs.push(s.slice(i+1, j).split(",")); i = j+1; }
    else { let t = ""; while (i < s.length && s[i] !== "{") t += s[i++]; segs.push(t); }
  }
  let res = [""];
  for (const seg of segs) { const opts = typeof seg === "string" ? [seg] : seg; res = res.flatMap(p => opts.map(o => p + o)); }
  return res.sort();
}

UnionFind
class UF {
  p = new Map<string, string>();
  find(x: string): string { if (!this.p.has(x)) this.p.set(x, x); if (this.p.get(x) !== x) this.p.set(x, this.find(this.p.get(x)!)); return this.p.get(x)!; }
  union(a: string, b: string) { this.p.set(this.find(a), this.find(b)); }
  connected(a: string, b: string) { return this.find(a) === this.find(b); }
}
topoSort
function topoSort(graph: Map<string, string[]>): string[] | null {
  const deg = new Map<string, number>();
  for (const [n, ns] of graph) { deg.set(n, deg.get(n) ?? 0); for (const x of ns) deg.set(x, (deg.get(x) ?? 0) + 1); }
  const q = [...deg].filter(([,d]) => d === 0).map(([n]) => n), out: string[] = [];
  while (q.length) { const n = q.shift()!; out.push(n); for (const x of graph.get(n) ?? []) { deg.set(x, deg.get(x)! - 1); if (deg.get(x) === 0) q.push(x); } }
  return out.length === deg.size ? out : null;
}
mergeIntervals
function mergeIntervals(ivs: [number, number][]): [number, number][] {
  const sorted = [...ivs].sort((a, b) => a[0] - b[0]);
  const res: [number, number][] = [sorted[0]];
  for (let i = 1; i < sorted.length; i++) {
    const last = res[res.length - 1];
    if (sorted[i][0] <= last[1]) last[1] = Math.max(last[1], sorted[i][1]);
    else res.push(sorted[i]);
  }
  return res;
}
HTTP / Integration
get / post
async function get<T>(url: string): Promise<T> { const r = await fetch(url); if (!r.ok) throw new Error(`${r.status}`); return r.json(); }
async function post<T>(url: string, body: unknown): Promise<T> {
  const r = await fetch(url, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify(body) });
  if (!r.ok) throw new Error(`${r.status}: ${await r.text()}`); return r.json();
}
paginateAll (cursor-based)
async function paginateAll<T extends { id: string }>(fetch_: (cursor?: string) => Promise<{ data: T[]; has_more: boolean }>) {
  const all: T[] = []; let cursor: string | undefined;
  while (true) { const p = await fetch_(cursor); all.push(...p.data); if (!p.has_more || !p.data.length) break; cursor = p.data.at(-1)!.id; }
  return all;
}
retry
async function retry<T>(fn: () => Promise<T>, n = 3, delay = 200): Promise<T> {
  for (let i = 0; i <= n; i++) { try { return await fn(); } catch (e) { if (i === n) throw e; await new Promise(r => setTimeout(r, delay * 2 ** i)); } }
  throw new Error();
}

import { readFileSync, writeFileSync } from "fs";
const readText = (p: string) => readFileSync(p, "utf-8");
const readLines = (p: string) => readText(p).trim().split("\n");
const readJSON = <T>(p: string): T => JSON.parse(readText(p));
const writeJSON = (p: string, d: unknown) => writeFileSync(p, JSON.stringify(d, null, 2));
